def _simplificar_coords(coords: list, max_pts: int = 80) -> list: # Reduzido de 120 para 80 """Simplificação adaptativa com limite máximo de pontos.""" if len(coords) <= max_pts: return coords try: from shapely.geometry import Polygon as SP poly = SP(coords) # Aumenta a tolerância para simplificar mais for tol in (0.001, 0.002, 0.005, 0.01, 0.02, 0.05): simp = poly.simplify(tol, preserve_topology=True) geom = simp if simp.geom_type == "Polygon" else max(simp.geoms, key=lambda g: g.area) c = list(geom.exterior.coords) if len(c) <= max_pts: return c except Exception: pass step = max(1, len(coords) // max_pts) return coords[::step][:max_pts]
